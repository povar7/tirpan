
    def test_walk_var_VAR(self):
        node = utils.findNode(self.ast, line=LINE, kind=ast.Name)
        self.assertTrue(node is not None, 'required node was not found')
        self.assertTrue(hasattr(node, 'link'), 'node has no link to type info')
        self.assertTrue(isinstance(node.link, VariableTGNode),
                        'type is not a var')
        nodeType = freezeSet(node.link.nodeType)
        type1 = TYPE
        type1.freeze()
        self.assertTrue(len(nodeType) == NUM and
                        any(type1 == elem for elem in nodeType),
                        'wrong types calculated')
        self.assertEqual(node.link.name, 'VAR', 'name is not "VAR"')
